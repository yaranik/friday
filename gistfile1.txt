#include <stdio.h>

int dayofweek(int d, int m, int y)
{
    static int t[] = { 0, 3, 2, 5, 0, 3, 5, 1, 4, 6, 2, 4 };
    y -= m < 3;
    return ( y + y/4 - y/100 + y/400 + t[m-1] + d) % 7;
}

int main() {
    int year = 1918;                        
    int friday = 0;
    while (year < 2167)
    {
        int month = 1;                       
        while (month <= 12)
        {
            int day = 1;                   
            while (day <= 31)
            {
                int daynow = dayofweek(day, month, year);
                if (day == 13 && daynow == 5)
                {
                    if (year == 1918 && month == 2 && day <= 15)
                    {}
                    else
                    {
                        printf ("\n %d.%d.%d", day, month, year);
                        friday++;
                    }
                }
            day++;
            }
        month++;
        }
    year++;
    }
    printf ("\n Общее количество тринадцатых пятниц : %d", friday);
}
